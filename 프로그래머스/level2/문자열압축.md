## 풀이
~~~java  
// 갯수대로 쪼개기
// 이전 단어랑 같으면 단어갯수 유지
// 이전 단어와 다르면 이전단어 초기화하고 단어갯수 추가(숫자갯수+문자수)
class Solution {
    public int solution(String s) {
        int answer = s.length();
        for(int i=1;i<s.length();i++){        
            String tmp = s.substring(0,i);
            int tmpLen = 0;
            int count = 1;
            
            for(int j=0;j<s.length()/i;j++){
                if(!tmp.equals(s.substring(i*j,i*j+i))){
                    if(count == 1){
                        tmpLen += i;
                    }else{
                        tmpLen += i + String.valueOf(count+1).length();
                    }
                    count = 1;
                    tmp = s.substring(i*j,i*j+i);
                }else if(j!=0){
                    count++;
                }
                
                // 마지막에
                if(j+1 == s.length()/i){
                    if(count == 1){
                        tmpLen += i;
                    }else{
                        tmpLen += i + String.valueOf(count+1).length();
                    }
                }
            }
            if(s.length()%i!=0){
                tmpLen += s.length()%i;                
            }

            if(answer > tmpLen) answer = tmpLen;
        }
        return answer;
    }
}
~~~  

테스트 6  
입력값 〉	"aaaaaaaaabaa"  
기댓값 〉	5  
실행 결과 〉	실행한 결괏값 6이 기댓값 5과 다릅니다.
